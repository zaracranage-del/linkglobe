PRODUCTS TABLE ──────────────────────────────────────────── CREATE TABLE IF NOT EXISTS products ( id BIGSERIAL PRIMARY KEY, handle TEXT NOT NULL, user_email TEXT NOT NULL, store TEXT NOT NULL DEFAULT 'brand', store_label TEXT NOT NULL DEFAULT 'Brand', emoji TEXT DEFAULT '🛍️', name TEXT NOT NULL DEFAULT 'Product', image TEXT, original_url TEXT NOT NULL, geo_link TEXT NOT NULL, created_at TIMESTAMPTZ DEFAULT NOW() ); -- ── 2. CLICKS TABLE (rich analytics) ──────────────────────────── -- Tracks every storefront click with full context CREATE TABLE IF NOT EXISTS clicks ( id BIGSERIAL PRIMARY KEY, handle TEXT NOT NULL, product_id BIGINT REFERENCES products(id) ON DELETE SET NULL, store TEXT, country TEXT DEFAULT 'Unknown', referrer TEXT DEFAULT 'Direct', device_type TEXT DEFAULT 'Desktop', os TEXT DEFAULT 'Unknown', browser TEXT DEFAULT 'Unknown', created_at TIMESTAMPTZ DEFAULT NOW() ); -- Index for fast analytics queries CREATE INDEX IF NOT EXISTS clicks_handle_idx ON clicks(handle); CREATE INDEX IF NOT EXISTS clicks_created_idx ON clicks(created_at DESC); CREATE INDEX IF NOT EXISTS clicks_product_idx ON clicks(product_id); -- ── 3. AFFILIATE PROGRAMS TABLE ────────────────────────────────── -- Creators connect their own affiliate programs — keep 100% commission CREATE TABLE IF NOT EXISTS affiliate_programs ( id BIGSERIAL PRIMARY KEY, handle TEXT NOT NULL, user_email TEXT NOT NULL, program_name TEXT NOT NULL, network TEXT NOT NULL DEFAULT 'other', store_domain TEXT, tracking_url TEXT, tag TEXT, notes TEXT, active BOOLEAN DEFAULT TRUE, created_at TIMESTAMPTZ DEFAULT NOW() ); -- ── 4. ENABLE ROW LEVEL SECURITY ──────────────────────────────── ALTER TABLE products ENABLE ROW LEVEL SECURITY; ALTER TABLE signups ENABLE ROW LEVEL SECURITY; ALTER TABLE clicks ENABLE ROW LEVEL SECURITY; ALTER TABLE affiliate_programs ENABLE ROW LEVEL SECURITY; -- ── 5. PRODUCTS POLICIES ──────────────────────────────────────── DROP POLICY IF EXISTS "Public read products" ON products; DROP POLICY IF EXISTS "Auth insert products" ON products; DROP POLICY IF EXISTS "Auth delete products" ON products; CREATE POLICY "Public read products" ON products FOR SELECT USING (true); CREATE POLICY "Auth insert products" ON products FOR INSERT WITH CHECK (auth.email() = user_email); CREATE POLICY "Auth delete products" ON products FOR DELETE USING (auth.email() = user_email); -- ── 6. CLICKS POLICIES ────────────────────────────────────────── DROP POLICY IF EXISTS "Public insert clicks" ON clicks; DROP POLICY IF EXISTS "Auth read clicks" ON clicks; -- Anyone can insert a click (storefront is public) CREATE POLICY "Public insert clicks" ON clicks FOR INSERT WITH CHECK (true); -- Only the handle owner can read their own clicks CREATE POLICY "Auth read clicks" ON clicks FOR SELECT USING ( handle IN ( SELECT handle FROM signups WHERE email = auth.email() ) ); -- ── 7. SIGNUPS POLICIES ───────────────────────────────────────── DROP POLICY IF EXISTS "Public read signups" ON signups; DROP POLICY IF EXISTS "Public insert signups" ON signups; DROP POLICY IF EXISTS "Auth update signups" ON signups; CREATE POLICY "Public read signups" ON signups FOR SELECT USING (true); CREATE POLICY "Public insert signups" ON signups FOR INSERT WITH CHECK (true); CREATE POLICY "Auth update signups" ON signups FOR UPDATE USING (auth.email() = email); -- ── 8. AFFILIATE PROGRAMS POLICIES ────────────────────────────── DROP POLICY IF EXISTS "Auth read programs" ON affiliate_programs; DROP POLICY IF EXISTS "Auth insert programs" ON affiliate_programs; DROP POLICY IF EXISTS "Auth update programs" ON affiliate_programs; DROP POLICY IF EXISTS "Auth delete programs" ON affiliate_programs; CREATE POLICY "Auth read programs" ON affiliate_programs FOR SELECT USING (auth.email() = user_email); CREATE POLICY "Auth insert programs" ON affiliate_programs FOR INSERT WITH CHECK (auth.email() = user_email); CREATE POLICY "Auth update programs" ON affiliate_programs FOR UPDATE USING (auth.email() = user_email); CREATE POLICY "Auth delete programs" ON affiliate_programs FOR DELETE USING (auth.email() = user_email); -- ── 9.
